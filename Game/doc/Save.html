def Save.update(session)
  db = ConnectDB.new()
  diff = nil
  case session.difficulty
  when :easy
    diff = 0
  when :medium
    diff = 1
  when :hard
    diff = 2
  when :random
    diff = 3
  end

  db.save(Connexion.getJoueur,session.to_yaml())
end
