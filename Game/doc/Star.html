def refresh(nbEtoile,nbEtoileObtenu)
        @stars.each{|child|
                @stars.remove(child)
        }
        (1.. nbEtoile).each { |e|
                       @star=Gtk::EventBox.new()
                       if(e <= nbEtoileObtenu)
                        sH=@ic.iconAsset.fetch(:starH)
                        sH.resize(75,75)
                        sH.applyOn(@star)
                else
                        sS=@ic.iconAsset.fetch(:starS)
                        sS.resize(75,75)
                        sS.applyOn(@star)
                end
                       @stars.add(@star)
               }
               @stars.show_all
end
