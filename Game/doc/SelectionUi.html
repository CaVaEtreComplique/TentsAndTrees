def update(newSelection)

        unselected = @selected.reject { |cell|
                newSelection.include?(cell)
        }
        unselect(unselected)

        newlySelected = newSelection.reject { |cell|
                @selected.include?(cell)
        }
        select(newlySelected)

        @selected = newSelection

end
