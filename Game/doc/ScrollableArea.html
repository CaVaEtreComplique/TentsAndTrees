def initialize(orientation)
  @gtkObject=Gtk::ScrolledWindow.new()
  @menu= Gtk::Box.new(orientation)
  @gtkObject.border_width=10
  @vp=Gtk::Viewport.new(nil,@gtkObject.vadjustment)
  @vp.add(@menu)
  @gtkObject.add(@vp)
end
