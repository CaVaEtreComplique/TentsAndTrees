def replay(continue)
        # Generation de la grille
        @gridPick = GridGenerator.new(@difficulty)
        # Generation de la partie
        if continue
                @time=@game.time+@gridPick.associatedTimer
        else
                setTime
                @score=0
        end
        @game=Game.new(self)

        self
end
